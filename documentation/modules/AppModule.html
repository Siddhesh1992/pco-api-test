import { Logger, Module } from '@nestjs/common';
import { AppController } from './app.controller';
import { AppService } from './app.service';
import { ConfigModule } from '@nestjs/config';
import { AwsModule } from './cloud_provider/aws/aws.module';
import { AwsService } from './cloud_provider/aws/service/aws.service';
import { AzureModule } from './cloud_provider/azure/azure.module';
import { AzureService } from './cloud_provider/azure/service/azure.service';
import { ScheduleModule } from '@nestjs/schedule';
import { HttpModule } from '@nestjs/axios';
import { DatabaseModule } from './database/database.module';
import { DatabaseService } from './database/database.service';
import { MigrationModule } from './migration/migration.module';
import { ServeStaticModule } from '@nestjs/serve-static';
import { resolve } from 'path';
import { KnexModule } from 'nestjs-knex';
import { knex_config } from './config';

const ENV = process.env.NODE_ENV;
@Module({
  imports: [
    KnexModule.forRoot(knex_config),
    ServeStaticModule.forRoot(
      (() => {
        const publicDir = resolve('./public');
        const servePath = '/public';

        return {
          rootPath: publicDir,
          // serveRoot - if you want to see files on another controller,
          // e.g.: http://localhost:8088/files/1.png
          serveRoot: servePath,
          exclude: ['/api*'],
        };
      })(),
    ),
    ConfigModule.forRoot({
      isGlobal: true,
      envFilePath: !ENV ? '.env' : `.env.${ENV}`,
    }),

    ScheduleModule.forRoot(),
    HttpModule,
    AwsModule,
    AzureModule,
    DatabaseModule,
    MigrationModule,
  ],
  controllers: [AppController],
  providers: [
    Logger,
    {
      provide: 'appService',
      useClass: AppService,
    },
    {
      provide: 'databaseService',
      useClass: DatabaseService,
    },
    {
      provide: 'awsService',
      useClass: AwsService,
    },
    {
      provide: 'azureService',
      useClass: AzureService,
    },
  ],
})
export class AppModule {}
