    @Query() query: DeleteServiceDto,
    @Body() body: PaginationDto,
    @Response({ passthrough: true }) res,
  ) {
    const { provider, service, download } = query;
    const { data, datatypes, count } = await this.moduleRef
      .get(`appService`, { strict: false })
      .getDeleteData({ data: { body }, provider, service });

    // Download File
    if (download == 'true') {
      const name = 'DeletedResources';
      const workbook = await this.moduleRef
        .get(`appService`, { strict: false })
        .downloadExcel({ data, name, datatypes });

      res.setHeader(
        'Content-Type',
        'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
      );
      res.setHeader(
        'Content-Disposition',
        'attachment; filename=' + name + '.xlsx',
      );
      await workbook.xlsx.write(res);
      res.status(200).end();
      return;
    }

    return {
      datatypes,
      count,
      data,
    };
  }
