import {
  BadRequestException,
  Body,
  Controller,
  Delete,
  Get,
  Param,
  Post,
  Query,
  Response,
  UploadedFile,
  UseInterceptors,
} from '@nestjs/common';
import { MigrationService } from './migration.service';
import {
  ApiBody,
  ApiConsumes,
  ApiParam,
  ApiQuery,
  ApiTags,
} from '@nestjs/swagger';
import { extname, join, resolve } from 'path';
import { ModuleRef } from '@nestjs/core';
import * as parser from 'cron-parser';
import * as moment from 'moment';
import { createReadStream } from 'fs';

@Controller('migration')
export class MigrationController {
  constructor(
    private readonly migrationService: MigrationService,
    private readonly moduleRef: ModuleRef,
  ) {}
  @ApiTags('migration')
  @ApiQuery({ name: 'frequency', example: '* 1 * * *' })
  @Get('/')
  async getSeedData(@Query() query) {
    try {
      console.log(query.frequency);
      const interval = parser.parseExpression(
        query?.frequency ? query.frequency : '* 1 * * *',
      );
      const next = interval.next();
      return {
        jsTime: new Date(),
        nextInterval: next.toDate(),
        momentTime: moment().format(),
        convertedJsTimeInMoment: moment(new Date()).format(),
      };
    } catch (e) {
      return 'error';
    }
  }

  @ApiTags('migration')
  @Delete('/delete-weekly-migration')
  async deleteSeed() {
    try {
      return this.migrationService.deleteWeeklySummaryData();
    } catch (e) {}
  }

  @Get('/rule')
  @ApiQuery({ name: 'type', required: false })
  async getRule(@Query() query) {
    try {
      const { type } = query;
      return this.migrationService.queryRuleEngine(type);
    } catch (e) {}
  }

  @Get('/label')
  async getLabel() {
    try {
      return this.migrationService.getLabel();
    } catch (e) {}
  }

  @Get('/ruleColumn')
  async ruleColumn() {
    try {
      return this.migrationService.ruleColumn();
    } catch (e) {}
  }

  @ApiQuery({ name: 'type', required: false })
  @Get('/logs')
  async getLogs(@Response() res, @Query() query: { type: string }) {
    try {
      const { type } = query;
      const file = createReadStream(
        join(process.cwd(), type ? `${type}.log` : 'combined.log'),
      );
      return file.pipe(res);
    } catch (e) {}
  }

  @ApiQuery({ name: 'process_id', required: false })
  @Delete('/update_processes_table_status_date')
  async update_table(@Query() query: { process_id: string }) {
    try {
      return this.migrationService.update_table(query);
    } catch (e) {}
  }

  @Get('/clearLogs')
  async clearLogs() {
    try {
      return this.migrationService.clearLogs();
    } catch (e) {}
  }

  @Get('/getCron')
  async getCron() {
    try {
      return this.migrationService.getCron();
    } catch (e) {}
  }

  @Get('/dbSeed')
  async dbSeed() {
    try {
      return this.migrationService.dbSeed();
    } catch (e) {}
  }
  @Get('/cloud-trail')
  async cloudTrails() {
    try {
      return this.migrationService.cloudTrials();
    } catch (e) {}
  }
}
