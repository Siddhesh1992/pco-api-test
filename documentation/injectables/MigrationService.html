import { Injectable } from '@nestjs/common';
import * as util from 'util';
import * as childProcess from 'child_process';
import { DatabaseService } from '../database/database.service';
import { InjectKnex, Knex } from 'nestjs-knex';
import { groupBy, omit } from 'lodash';
import { readFileSync, unlink, writeFile } from 'fs';
import {
  CloudWatchClient,
  GetMetricDataCommand,
  GetMetricDataCommandInput,
} from '@aws-sdk/client-cloudwatch'; // ES Modules import
import {
  CloudTrailClient,
  AddTagsCommand,
  LookupEventsCommand,
} from '@aws-sdk/client-cloudtrail';
import { forEach } from 'lodash';
// const { CloudWatchClient, GetMetricDataCommand } = require("@aws-sdk/client-cloudwatch"); // CommonJS import

import * as moment from 'moment';
import { ModuleRef } from '@nestjs/core';
const exec = util.promisify(childProcess.exec);

@Injectable()
export class MigrationService {
  constructor(
    private readonly db: DatabaseService,
    @InjectKnex() private readonly query: Knex,
    private readonly moduleRef: ModuleRef,
  ) {}

  async deleteWeeklySummaryData() {
    await this.db.weekly_tracker.deleteMany();
    try {
      await this.db.$queryRaw`TRUNCATE weekly_tracker RESTART IDENTITY;`;
    } catch (e) {}
  }

  async insertMany(obj: []) {
    await this.db.weekly_tracker.createMany({
      data: obj,
    });
    try {
      await this.db.$queryRaw`TRUNCATE weekly_tracker RESTART IDENTITY;`;
    } catch (e) {}
  }

  async getRule(rule: string) {
    return this.db.rule.findFirst({
      where: {
        name: rule,
      },
      select: {
        provider: true,
        select: true,
        group: true,
        order: true,
        service: true,
        condition: {
          select: {
            id: true,
            attribute: true,
            operator: true,
            value: true,
            cond: true,
            preference: true,
          },
          orderBy: {
            preference: 'asc',
          },
        },
      },
    });
  }

  async queryRuleEngine(type) {
    const rules = await this.getRule(type);
    const table = `${rules.provider}_${rules.service}`;

    const queryBuilder = this.query.table(table);
    const groupedCondtion = groupBy(rules.condition, 'preference');

    for (const condition in groupedCondtion) {
      const rows = groupedCondtion[condition];
      console.log(rows, 'rows');
      if (rows.length == 1) {
        for (const row of rows) {
          if (row.cond == 'AND') {
            queryBuilder.where(row.attribute, row.operator, row.value);
          } else if (row.cond == 'OR') {
            queryBuilder.orWhere(row.attribute, row.operator, row.value);
          }
        }
      } else {
        if (rows[0].cond == 'AND') {
          queryBuilder.andWhere((query) => {
            for (const groupRow of rows) {
              if (groupRow.cond == 'AND') {
                query.where(
                  groupRow.attribute,
                  groupRow.operator,
                  groupRow.value,
                );
              } else if (groupRow.cond == 'OR') {
                query.orWhere(
                  groupRow.attribute,
                  groupRow.operator,
                  groupRow.value,
                );
              }
            }
          });
        } else if (rows[0].cond == 'OR') {
          queryBuilder.orWhere((query) => {
            for (const groupRow of rows) {
              if (groupRow.cond == 'AND') {
                query.where(
                  groupRow.attribute,
                  groupRow.operator,
                  groupRow.value,
                );
              } else if (groupRow.cond == 'OR') {
                query.orWhere(
                  groupRow.attribute,
                  groupRow.operator,
                  groupRow.value,
                );
              }
            }
          });
        }
      }
    }
    console.log(rules['select'], 'rules');
    if (rules['select']) {
      queryBuilder.select(this.query.raw(rules['select']));
    }

    if (rules.group) {
      queryBuilder.groupByRaw(rules.group);
    }

    if (rules.order) {
      queryBuilder.orderByRaw(rules.order);
    }
    console.log(queryBuilder.toString());
    return { query: queryBuilder.toString(), data: await queryBuilder };
  }

  async ruleColumn() {
    const columns = await this.query.table('aws_rds').first();

    return Object.keys(
      omit(columns, ['id', 'deleted', 'created_at', 'updated_at']),
    );
  }

  async getLabel() {
    return this.db.label.findMany();
  }

  async getLogs(query: { type: string }) {
    const { type } = query;
    console.log(`./${type ? type : 'error'}.log`);
    return readFileSync(`./${type ? type : 'error'}.log`, 'utf8');
  }

  async clearLogs() {
    try {
      writeFile('./combined.log', '', function () {
        console.log('done');
      });
      writeFile('./error.log', '', function () {
        console.log('done');
      });
    } catch (e) {
      console.log(e);
    }
  }

  async getCron() {
    return this.db.processes.findMany({
      orderBy: {
        id: 'asc',
      },
    });
  }

  async resetCronStatus() {
    try {
      return this.db.processes.updateMany({
        data: {
          status: 'pending',
        },
      });
    } catch (e) {}
  }

  async dbSeed() {
    try {
      const { stdout, stderr } = await exec('npx prisma db seed');
    } catch (e) {}
  }

  async deleteCron() {
    try {
      await this.db.rule_condition.deleteMany();
      await this.db.rule.deleteMany();
      await this.db.$queryRaw`TRUNCATE processes RESTART IDENTITY cascade`;
      await this.db.$queryRaw`TRUNCATE rule_condition RESTART IDENTITY cascade`;
      await this.db.$queryRaw`TRUNCATE rule RESTART IDENTITY cascade`;
    } catch (e) {
      console.log(e);
    }
  }

  async queryProcess() {
    const cronProcess = await this.query('processes')
      .select(this.query.raw(`Max(id) as max_id`))
      .first();
    return cronProcess;
  }

  async update_table(query: { process_id: string }) {
    const { process_id } = query;
    console.log(moment().subtract(1, 'days').toDate(), 'moment');
    return this.db.processes.update({
      where: {
        id: Number(process_id),
      },
      data: {
        date_time: moment().subtract(2, 'days').toDate(),
        status: 'pending',
      },
    });
  }

  async cloudTrials() {
    console.log(moment().subtract(1, 'h').toDate());
    try {
      const options = {
        LookupAttributes: [
          {
            AttributeKey: 'EventName',
            AttributeValue: 'DetachVolume',
          },
        ],
        StartTime: moment().subtract(3, 'h').toDate(),
        EndTime: moment().toDate(),
        // EventCategory: 'insight',
        MaxResults: 50,
      };
      let account;
      const region = 'us-east-1';
      const accounts = JSON.parse(process.env.AWS_ACCOUNTS);

      //get account
      forEach(accounts, (accountNumber, alias) => {
        if (alias == 'viacbs-app-dev') {
          account = { alias, accountNumber };
          return false;
        }
      });

      if (!account) {
        return {
          data: null,
          error: `Account not found for ${JSON.stringify(account)}`,
        };
      }

      const response = await this.moduleRef
        .get(`appService`, { strict: false })
        .assumeRole(account);

      if (!response) {
        return {
          data: null,
          error: `Cannot assume role for
          )}`,
        };
      }

      const client = new CloudTrailClient({
        region,
        credentials: {
          accessKeyId: response.Credentials.AccessKeyId,
          secretAccessKey: response.Credentials.SecretAccessKey,
          sessionToken: response.Credentials.SessionToken,
        },
      });

      const command = new LookupEventsCommand(options);
      const response1 = await client.send(command);
      console.log(response1);
    } catch (e) {
      console.log(e);
    }
  }
}
