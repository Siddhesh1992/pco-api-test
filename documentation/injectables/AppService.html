  async livenessProbe() {
    try {

      throw new InternalServerErrorException(error.message, error.status);

  async readinessProbe() {
    try {
      // Try and access database
      const count = await this.db.processes.count();
      if (count > 0) {
        return { message: 'Database is running' };
      } else {
        throw new InternalServerErrorException('Database is not configured');
      }
    } catch (error) {
      throw new InternalServerErrorException(error.message, error.status);
    }
  }

  async verifyToken(options: {
    token: string;
    token_type_hint: string;
  }): Promise<any> {
    try {
      const response = await firstValueFrom(
        this.httpService.post(
          process.env.OKTA_INTROSPECT_URL,
          new URLSearchParams(options),
          {
            headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
            params: { client_id: process.env.OKTA_CLIENT_ID },
          },
        ),
      );
      return has(response.data, 'active') ? response.data.active : false;
    } catch (error) {
      this.logger.error(`Verify Token: ${error.message}`);
    }
  }

  async getUserInfo(token): Promise<any> {
    try {
      const response = await firstValueFrom(
        this.httpService.get(process.env.OKTA_USERINFO_URL, {
          headers: {
            'Content-Type': 'application/json',
            Authorization: token,
          },
        }),
      );
      return response;
    } catch (error) {
      this.logger.error(`Get User Info: ${error.message}`);
    }
  }

  async assumeRole(account: { alias: string; accountNumber: string }) {
    try {
      // Initialization of Client
      const client = new STS({ region: 'us-east-1' });

      const params = {
        RoleArn: `arn:aws:iam::${account.accountNumber}:role/${process.env.AWS_ASSUME_ROLE}`,
        RoleSessionName: 'AssumeRoleSession1',
      };
      const command = new AssumeRoleCommand({
        ...params,
        // DurationSeconds: 43200,
      });
      const response = await client.send(command);
      return response;
    } catch (error) {
      this.logger.error(
        `Assuming Role for (${account.alias} with ${process.env.AWS_ASSUME_ROLE}) : ${error.message}`,
      );
    }
  }

  async checkProcessStatus(name: string) {
    try {
      // Check process status
      const record = await this.db.processes.findFirst({
        where: {
          name: name,
          status: 'in-progress',
          disabled: false,
          deleted: null,
          start_time: {
            gte: new Date(moment().format('YYYY-MM-DD').toString()),
          },
        },
      });
      return record;
    } catch (error) {
      this.logger.error(`CheckProcessStatus: ${error.message}`);
    }
  }

    }
  }

  async weeklyCostSavingsTracker(options: { id: number; year: string }) {
    try {
      const weeks = [...Array(52).keys()].map((i) => i + 1);
      const response = [];

      const targetDetails = await this.getMiscellaneousData('weekly_target');
      // Query to Get Data
      const savingsQuery: any = await this.db.$queryRaw(Prisma.sql`SELECT
      EXTRACT(YEAR FROM "date_of_implementation") as "year",
      DATE_PART('Week',date_of_implementation) as "week",
      'Week'||DATE_PART('Week',date_of_implementation) as "name",
      SUM("cost_saved") as "savings" 
      FROM public.weekly_tracker 
      WHERE provider_id= ${options.id} AND DATE_PART('year', date_of_implementation)::VARCHAR = ${options.year}
      GROUP BY 1,2 ORDER BY 2`);

      // Get Min and Max Values
      let minSavings = 0;
      let maxSavings = 0;
      let minWeek = 0;
      let maxWeek = 0;

      if (savingsQuery.length > 0) {
        const values = savingsQuery.map((value) => value['savings']);
        minSavings = round(Math.min.apply(null, values), 2);
        maxSavings = round(Math.max.apply(null, values), 2);

        const weekValues = savingsQuery.map((value) => value['week']);
        minWeek = Math.min.apply(null, weekValues);
        maxWeek = Math.max.apply(null, weekValues);
      }

      let count = 1;
      let cumulativeSavingsForeCast = 0;
      let cumulativeSavings = 0;
      let forecastValue = 0;
      const movingAverage = [];

      for (const week of weeks) {
        const name = `Week${week}`;
        let costSavedDetails = savingsQuery.find((data) => data.name === name);
        const details = { week: name, type: 'weekly', year: options.year };

        // Get last 3 savings to calculate moving average
        const costSavings = {
          first: savingsQuery.find((data) => data.week === week),
          second: savingsQuery.find((data) => data.week === week - 1),
          third: savingsQuery.find((data) => data.week === week - 2),
        };

        if (week < maxWeek + 1) {
          movingAverage[0] = costSavings['first']
            ? costSavings['first']['savings']
            : 0;
          movingAverage[1] = costSavings['second']
            ? costSavings['second']['savings']
            : 0;
          movingAverage[2] = costSavings['third']
            ? costSavings['third']['savings']
            : 0;
        } else {
          movingAverage.unshift(round(sum(movingAverage) / 3, 2));
          movingAverage.pop();
        }

        // Check if Savings Present
        if (has(costSavedDetails, 'savings')) {
          details['savings'] = round(costSavedDetails.savings, 2);
          cumulativeSavings += details['savings'];
          details['cumulativeSavings'] = round(cumulativeSavings, 2);
        } else if (
          count > minWeek &&
          count < maxWeek &&
          !has(costSavedDetails, 'savings')
        ) {
          details['savings'] = 0;
          cumulativeSavings += details['savings'];
          details['cumulativeSavings'] = round(cumulativeSavings, 2);
        }

        if (savingsQuery.length > 0) {
          if (
            count > minWeek &&
            count < maxWeek &&
            !has(costSavedDetails, 'savings')
          ) {
            details['savingsForeCast'] = 0;
          } else {
            details['savingsForeCast'] = has(costSavedDetails, 'savings')
              ? ''
              : movingAverage[0];
          }
        } else {
          details['savingsForeCast'] = 0;
        }

        details['target'] = has(targetDetails, 'value')
          ? parseInt(targetDetails['value'])
          : 0;
        details['cumulativeTarget'] = details['target'] * count;
        cumulativeSavingsForeCast += toNumber(details['savingsForeCast']);

        if (details['savings'] >= 0) {
          forecastValue += details['savings'];
        } else {
          forecastValue += details['savingsForeCast'];
        }

        details['forecast'] =
          details['savingsForeCast'] !== '' ? round(forecastValue, 2) : '';

        count++;
        response.push(details);
      }

      return response;
    } catch (error) {
      this.logger.error(`WeeklyCostSavingsTracker : ${error.message}`);
    }
  }

  /**
   * Create a Rule
   *
   * @param body RuleDto
   * @param token A user token
   * @returns A promise with created all rules
   */
  async createRule(body: RuleDto, token: any) {
    try {
      const user = await this.userInfo(token);

      if (!user?.email) {
        throw new NotFoundException('Email not found');
      }

      const { rules, name } = body;

      const provider = body.provider.toLowerCase();
      const service = body.service.toLowerCase();

      const getRule = await this.db.rule.findFirst({
        where: {
          provider,
          service,
          name,
          // created_by: user.email,
          deleted: null,
        },
      });

      if (getRule) {
        throw new BadRequestException('Rule already defined');
      }
      if (rules && rules.length == 0) {
        throw new BadRequestException(
          'There should be atleast one rule defined',
        );
      }

      const columns = await this.getAttributes({
        provider: provider,
        service: service,
      });

      for (const condition of rules) {
        await this.validateOperators(condition, columns);
      }

      const ruleCondition = [];
      for (const condition of rules) {
        const {
          attribute,
          comparisonOperator,
          logicalOperator,
          preference,
          value,
        } = condition;
        const modifiedValue = await this.getModifiedValue(
          comparisonOperator,
          value,
        );

        ruleCondition.push({
          attribute: attribute.toLowerCase(),
          cond: logicalOperator.toUpperCase(),
          operator: ['startsWith', 'endsWith', 'contains'].includes(
            comparisonOperator,
          )
            ? 'ilike'
            : comparisonOperator,
          preference,
          value: modifiedValue,
        });
      }

      await this.db.rule.create({
        select: {
          id: true,
        },
        data: {
          name,
          provider,
          service,
          created_by: user?.email,
          condition: {
            createMany: {
              data: ruleCondition,
            },
          },
        },
      });

      return { message: 'success', data: [] };
    } catch (err) {
      this.logger.error(`createRule : ${err.message}`);
      throw err;
    }
  }

  /**
   * A Rule condition is built by a Library KnexJS to get sql query
   *
   * @param groupedCondtion A Condition having same preference will be grouped as one condition
   * @param queryBuilder A knexJS query builder object for adding condition
   * @returns null
   */
  async buildWhereQuery(groupedCondtion, queryBuilder) {
    for (const condition in groupedCondtion) {
      const rows = groupedCondtion[condition];
      if (rows.length == 1) {
        for (const row of rows) {
          if (row.cond == 'AND') {
            if (['NOW()', 'CURRENT_DATE', 'NULL'].includes(row.value)) {
              queryBuilder.whereRaw(
                `${row.attribute} ${row.operator} ${row.value}`,
              );
            } else {
              queryBuilder.where(row.attribute, row.operator, row.value);
            }
          } else if (row.cond == 'OR') {
            if (['NOW()', 'CURRENT_DATE', 'NULL'].includes(row.value)) {
              queryBuilder.orWhereRaw(
                `${row.attribute} ${row.operator} ${row.value}`,
              );
            } else {
              queryBuilder.orWhere(row.attribute, row.operator, row.value);
            }
          }
        }
      } else {
        if (rows[0].cond == 'AND') {
          queryBuilder.andWhere((query) => {
            for (const groupRow of rows) {
              if (groupRow.cond == 'AND') {
                if (
                  ['NOW()', 'CURRENT_DATE', 'NULL'].includes(groupRow.value)
                ) {
                  query.whereRaw(
                    `${groupRow.attribute} ${groupRow.operator} ${groupRow.value}`,
                  );
                } else {
                  query.where(
                    groupRow.attribute,
                    groupRow.operator,
                    groupRow.value,
                  );
                }
              } else if (groupRow.cond == 'OR') {
                if (
                  ['NOW()', 'CURRENT_DATE', 'NULL'].includes(groupRow.value)
                ) {
                  query.orWhereRaw(
                    `${groupRow.attribute} ${groupRow.operator} ${groupRow.value}`,
                  );
                } else {
                  query.orWhere(
                    groupRow.attribute,
                    groupRow.operator,
                    groupRow.value,
                  );
                }
              }
            }
          });
        } else if (rows[0].cond == 'OR') {
          queryBuilder.orWhere((query) => {
            for (const groupRow of rows) {
              if (groupRow.cond == 'AND') {
                if (
                  ['NOW()', 'CURRENT_DATE', 'NULL'].includes(groupRow.value)
                ) {
                  query.whereRaw(
                    `${groupRow.attribute} ${groupRow.operator} ${groupRow.value}`,
                  );
                } else {
                  query.where(
                    groupRow.attribute,
                    groupRow.operator,
                    groupRow.value,
                  );
                }
              } else if (groupRow.cond == 'OR') {
                if (
                  ['NOW()', 'CURRENT_DATE', 'NULL'].includes(groupRow.value)
                ) {
                  query.orWhereRaw(
                    `${groupRow.attribute} ${groupRow.operator} ${groupRow.value}`,
                  );
                } else {
                  query.orWhere(
                    groupRow.attribute,
                    groupRow.operator,
                    groupRow.value,
                  );
                }
              }
            }
          });
        }
      }
    }
  }

  /**
   * Compile Rule & condtion and get a sql query with data
   *
   * @param where A filter object
   * @param onlyQuery To get only Query or excute the query and get data too
   * @returns A query along with rule data if onlyQuery param is false
   */
  async queryRule(where: any, onlyQuery = false, options = {}) {
    try {
      const rules = await this.getRule(where);
      if (!rules) {
        throw new NotFoundException('Rule not found');
      }
      const table = `${rules.provider}_${rules.service}`;
      const columns = await this.query(table).columnInfo();
      const queryWrapper = this.query(table);
      const groupedCondtion = groupBy(rules.condition, 'preference');
      let recent = { data_refresh_date_time: '' };
      if (!onlyQuery)
        if (columns['data_refresh_date_time']) {
          recent = await getLatestRefreshDate(table);
        }
      if (columns['data_refresh_date_time'] && recent?.data_refresh_date_time) {
        // we don't want brackets for one condition
        if (Object.keys(groupedCondtion).length > 1) {
          queryWrapper.andWhere((query) => {
            this.buildWhereQuery(groupedCondtion, query);
            //paginate logic
          });
        } else {
          this.buildWhereQuery(groupedCondtion, queryWrapper);
          //paginate logic
        }

        if (Object.keys(options)) {
          queryWrapper.andWhere((query) => {
            applyWhereFilter(query, options, this.query);
          });
        }

        queryWrapper.andWhere(
          'data_refresh_date_time',
          '>=',
          moment(recent.data_refresh_date_time).format('YYYY-MM-DD'),
        );
      } else {
        this.buildWhereQuery(groupedCondtion, queryWrapper);
        //paginate logic
        if (Object.keys(options)) {
          queryWrapper.andWhere((query) => {
            applyWhereFilter(query, options, this.query);
          });
        }
      }

      if (rules['select']) {
        queryWrapper.select(this.query.raw(rules['select']));
      }

      if (rules.group) {
        queryWrapper.groupByRaw(rules.group);
      }

      if (rules.having) {
        queryWrapper.havingRaw(rules.having);
      }

      if (rules.order) {
        queryWrapper.orderByRaw(rules.order);
      }

      if (Object.keys(options))
        applyPaginationFilter(queryWrapper, options, this.query);

      const query = queryWrapper.toString();
      const data = onlyQuery ? [] : await queryWrapper;
      return { message: query, data };
    } catch (err) {
      this.logger.error(`queryRule : ${err?.message}`);
      throw err;
    }
  }

  async incrementScheduler(cronProcess) {
    try {
      const currentDate = moment().toDate();
      const processDate = new Date(cronProcess.date_time);

      const interval = parser.parseExpression(cronProcess.frequency, {
        currentDate: processDate,
        iterator: true,
      });

      let next = interval.next();

      this.logger.log(
        `incrementing the processes ${
          cronProcess.id
        } with process date = ${processDate} next date = ${next.value.toDate()}`,
      );

      while (currentDate >= next.value.toDate()) {
        this.logger.log(
          `incrementing the processes while loop ${
            cronProcess.id
          } with process date = ${processDate} next date = ${next.value.toDate()}`,
        );
        next = interval.next();
      }

      // cron started early
      while (processDate >= next.value.toDate()) {
        this.logger.log(
          `incrementing the processes while loop ${
            cronProcess.id
          } with process date = ${processDate} next date = ${next.value.toDate()}`,
        );
        next = interval.next();
      }

      this.logger.log(
        `incremented the processes ${
          cronProcess.id
        } to ${next.value.toDate()} with process date = ${processDate}`,
      );
      await this.db.processes.update({
        where: {
          id: cronProcess.id,
        },
        data: {
          end_time: null,
          status: 'pending',
          date_time: next.value.toDate(),
        },
      });
    } catch (e) {
      this.logger.error(
        `Error incrementing date for ${cronProcess.id} ${e.mesage}`,
      );
    }
  }

  async ruleScheduler(query: { process_id: string; queue_id: string }) {
    try {
      setTimeout(async () => {
        try {
          const { process_id, queue_id } = query;
          const data = [];

          this.logger.log(
            `started process for cron ${process_id} ${moment().toLocaleString()}`,
          );

          // get current process
          const cronProcess = await this.getProcess({ id: Number(process_id) });
          // update current process as in-process
          await this.db.processes.update({
            where: {
              id: cronProcess.id,
            },
            data: {
              start_time: moment().format(),
              status: 'in-progress',
            },
          });

          //get all rule with data
          const response = await this.queryRule({
            id: Number(cronProcess.rule_id),
          });

          //save to label table
          if (response?.data && response?.data.length) {
            for (const result of response.data) {
              data.push({
                processes_id: cronProcess.id,
                type: cronProcess.type,
                resource_id: String(result.id),
                on_date: new Date(),
              });
            }

            await this.db.label.createMany({
              data,
            });

            this.logger.log(
              `completed process for cron ${process_id} ${moment().toLocaleString()}`,
            );
          }

          await this.db.processes.update({
            where: {
              id: cronProcess.id,
            },
            data: {
              end_time: moment().format(),
              status: 'pending',
            },
          });
        } catch (err) {
          const { process_id, queue_id } = query;
          this.logger.log(
            `Error process for cron ${process_id} ${err.toString()} ${moment().toLocaleString()}`,
          );
          await this.resetScheduler(process_id);
        }
      }, 5000);
      return { message: 'Rule Scheduled' };
    } catch (e) {}
  }

  async generateFilters(filters: Object) {
    try {
      const conditions = [];
      for (const column in filters) {
        const fieldContraints = filters[column];
        const operator = fieldContraints['operator'].toUpperCase();
        let condition = { [operator]: [] };

        for (const constraint of fieldContraints['constraints']) {
          if (constraint['matchMode'] === 'contains') {
            condition[operator].push({
              [column]: {
                contains: constraint['value'],
                mode: 'insensitive',
              },
            });
          } else if (constraint['matchMode'] === 'notContains') {
            condition[operator].push({
              NOT: {
                [column]: {
                  contains: constraint['value'],
                  mode: 'insensitive',
                },
              },
            });
          } else if (constraint['matchMode'] === 'endsWith') {
            condition[operator].push({
              [column]: {
                endsWith: constraint['value'],
                mode: 'insensitive',
              },
            });
          } else if (constraint['matchMode'] === 'startsWith') {
            condition[operator].push({
              [column]: {
                startsWith: constraint['value'],
                mode: 'insensitive',
              },
            });
          } else if (constraint['matchMode'] === 'notEquals') {
            condition[operator].push({
              [column]: {
                not: constraint['value'],
                ...(Number(constraint['value']) >= 0
                  ? {}
                  : { mode: 'insensitive' }),
              },
            });
          } else if (constraint['matchMode'] === 'equals') {
            condition[operator].push({
              [column]: {
                equals: constraint['value'],
                ...(Number(constraint['value']) >= 0
                  ? {}
                  : { mode: 'insensitive' }),
              },
            });
          } else if (constraint['matchMode'] === 'lt') {
            condition[operator].push({
              [column]: {
                lt: constraint['value'],
              },
            });
          } else if (constraint['matchMode'] === 'lte') {
            condition[operator].push({
              [column]: {
                lte: constraint['value'],
              },
            });
          } else if (constraint['matchMode'] === 'gt') {
            condition[operator].push({
              [column]: {
                gt: constraint['value'],
              },
            });
          } else if (constraint['matchMode'] === 'gte') {
            condition[operator].push({
              [column]: {
                gte: constraint['value'],
              },
            });
          } else if (constraint['matchMode'] === 'dateAfter') {
            condition[operator].push({
              [column]: {
                gt: new Date(constraint['value']),
              },
            });
          } else if (constraint['matchMode'] === 'dateBefore') {
            condition[operator].push({
              [column]: {
                lt: new Date(constraint['value']),
              },
            });
          } else if (constraint['matchMode'] === 'dateIsNot') {
            condition[operator].push({
              [column]: {
                not: new Date(constraint['value']),
              },
            });
          } else if (constraint['matchMode'] === 'dateIs') {
            const ltDate = moment(constraint['value'], 'YYYY-MM-DD')
              .add(1, 'day')
              .format('YYYY-MM-DD');
            condition[operator].push({
              [column]: {
                gte: new Date(constraint['value']),
                lt: new Date(new Date(ltDate)),
              },
            });
          } else if (constraint['matchMode'] === 'isNotNull') {
            condition[operator].push({
              [column]: {
                not: null,
              },
            });
          } else if (constraint['matchMode'] === 'isNull') {
            condition[operator].push({
              [column]: {
                equals: null,
              },
            });
          }
        }
        conditions.push(condition);
      }

      return conditions;
    } catch (error) {
      this.logger.error(`GenerateFilters: ${error.message}`);
    }
  }

  async validateOperators(rule, columns) {
    try {
      await this.db
        .$queryRaw`SELECT setval('rule_id_seq', (SELECT MAX(id) from rule));`;
    } catch (e) {}

    try {
      await this.db
        .$queryRaw`SELECT setval('rule_condition_id_seq', (SELECT MAX(id) from rule_condition));`;
    } catch (e) {}
    if (!columns[rule.attribute]) {
      throw new NotFoundException('Invalid Attribute ' + rule.attribute);
    }

    if (
      ![
        '>',
        '<',
        '=',
        '!=',
        '>=',
        '<=',
        'startsWith',
        'endsWith',
        'contains',
        'IS',
        'IS NOT',
      ].includes(rule.comparisonOperator)
    ) {
      throw new NotFoundException(
        'Invalid comparison Operator ' + rule.comparisonOperator,
      );
    }

    if (!['AND', 'OR'].includes(rule.logicalOperator.toUpperCase())) {
      throw new NotFoundException(
        'Invalid Logical Operator ' + rule.logicalOperator,
      );
    }
  }

  async getModifiedValue(comparisonOperator, value) {
    const cmpOp = comparisonOperator.toLowerCase();
    let modifiedValue;
    try {
      if (cmpOp == 'startsWith') {
        modifiedValue = `${value}%`;
      } else if (cmpOp == 'endsWith') {
        modifiedValue = `%${value}`;
      } else if (cmpOp == 'contains') {
        modifiedValue = `%${value}%`;
      } else if (cmpOp == 'is' || cmpOp == 'is not') {
        modifiedValue = 'NULL';
      } else {
        modifiedValue = value;
      }
    } catch {
      this.logger.error('invalid Rule');
    }
    return modifiedValue;
  }

  async getConditionFormatedEdit(condition) {
    try {
      return condition.map((res) => {
        const {
          attribute,
          operator,
          cond: logicalOperator,
          value: modifiedValue,
          preference,
        } = res;

        const regex = /%/g;
        const match = [...modifiedValue.matchAll(regex)];

        let comparisonOperator = operator;
        let value = modifiedValue;
        if (match.length) {
          if (match.length == 2) {
            comparisonOperator = 'contains';
            value = modifiedValue.replace(regex, '');
          } else if (value.indexOf('%') == 0) {
            comparisonOperator = 'startsWith';
            value = modifiedValue.replace(regex, '');
          } else {
            comparisonOperator = 'endsWith';
            value = modifiedValue.replace(regex, '');
          }
        }

        return {
          attribute,
          comparisonOperator,
          logicalOperator,
          value,
          preference,
        };
      });
    } catch (e) {
      this.logger.error(e);
      return [];
    }
  }

        where: {
          id: permission['id'],
        },
        include: {
          pages: {
            select: {
              page_id: true,
            },
          },
